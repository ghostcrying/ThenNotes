```

filter.setValue(ciImage, forKey: kCIInputImageKey)
filter.setValue(0.0, forKey: kCIInputBrightnessKey) // 亮度 -1-1, 默认即可
filter.setValue(1.0, forKey: kCIInputContrastKey)   // 对比度0-4, 默认即可
filter.setValue(2.0, forKey: kCIInputSaturationKey) // 设置饱和度 0-2
```
